	id INTEGER PRIMARY KEY AUTOINCREMENT,

    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,
    dayOfWeek INTEGER NOT NULL,
    dayOfMonth INTEGER NOT NULL
);

    dayOfAccess TEXT NOT NULL
