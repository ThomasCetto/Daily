CREATE TABLE task(
	id INTEGER PRIMARY KEY AUTOINCREMENT,
	name TEXT NOT NULL,
	day TEXT NOT NULL,
	important INTEGER  NOT NULL,
	checked INTEGER NOT NULL
);

CREATE TABLE repetitive(
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,
    dayOfWeek INTEGER NOT NULL,
    dayOfMonth INTEGER NOT NULL
);
